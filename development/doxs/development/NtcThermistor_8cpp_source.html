151template <typename AdcType>
152ntc_err_t NtcThermistor<AdcType>::ReadTemperatureCelsius(float* temperature_celsius) noexcept {
153 if (temperature_celsius == nullptr) {
154 return NTC_ERR_NULL_POINTER;
155 }
156
157 if (!initialized_) {
158 return NTC_ERR_NOT_INITIALIZED;
159 }
160
161 // Read voltage from ADC
162 float voltage_volts = 0.0f;
163 ntc_err_t voltage_error = readAdcVoltage(&voltage_volts);
164 if (voltage_error != NTC_SUCCESS) {
165 return voltage_error;
166 }
167
168 // Calculate resistance
169 float resistance_ohms = 0.0f;
170 ntc_err_t resistance_error = calculateResistance(voltage_volts, &resistance_ohms);
171 if (resistance_error != NTC_SUCCESS) {
172 return resistance_error;
173 }
174
175 // Convert resistance to temperature
176 float raw_temperature = 0.0f;
177 ntc_err_t conversion_error = convertResistanceToTemperature(resistance_ohms, &raw_temperature);
178 if (conversion_error != NTC_SUCCESS) {
179 return conversion_error;
180 }
181
182 // Apply calibration offset
183 *temperature_celsius = raw_temperature + config_.calibration_offset;
184
185 // Apply filtering if enabled
186 if (config_.enable_filtering) {
187 *temperature_celsius = applyFiltering(*temperature_celsius);
188 }
189
190 // Validate temperature range
191 if (!NTC::ValidateTemperature(*temperature_celsius, config_.min_temperature,
192 config_.max_temperature)) {
193 return NTC_ERR_TEMPERATURE_OUT_OF_RANGE;
194 }
195
196 return NTC_SUCCESS;
197}

228template <typename AdcType>
229ntc_err_t NtcThermistor<AdcType>::ReadTemperature(ntc_reading_t* reading) noexcept {
230 if (reading == nullptr) {
231 return NTC_ERR_NULL_POINTER;
232 }
233
234 // Initialize reading structure
235 reading->error = NTC_SUCCESS;
236 reading->is_valid = false;
237 reading->timestamp_us = 0; // Implementer should provide timestamp
238 reading->accuracy_celsius = 0.5F; // Estimate based on typical NTC accuracy
239
240 // Read temperature
241 float temperature_celsius = 0.0f;
242 ntc_err_t error = ReadTemperatureCelsius(&temperature_celsius);
243
244 if (error == NTC_SUCCESS) {
245 // Get additional data
246 float resistance_ohms = 0.0f;
247 GetResistance(&resistance_ohms);
248
249 float voltage_volts = 0.0f;
250 GetVoltage(&voltage_volts);
251
252 uint32_t adc_raw = 0U;
253 GetRawAdcValue(&adc_raw);
254
255 // Fill reading structure
256 reading->temperature_celsius = temperature_celsius;
257 reading->temperature_fahrenheit = CelsiusToFahrenheit(temperature_celsius);
258 reading->temperature_kelvin = CelsiusToKelvin(temperature_celsius);
259 reading->resistance_ohms = resistance_ohms;
260 reading->voltage_volts = voltage_volts;
261 reading->adc_raw_value = adc_raw;
262 reading->is_valid = true;
263 }
264
265 reading->error = error;
266 return error;
267}

305template <typename AdcType>
306ntc_err_t NtcThermistor<AdcType>::GetRawAdcValue(uint32_t* adc_value) noexcept {
307 if (adc_value == nullptr) {
308 return NTC_ERR_NULL_POINTER;
309 }
310
311 if (!initialized_) {
312 return NTC_ERR_NOT_INITIALIZED;
313 }
314
315 if (adc_interface_ == nullptr) {
316 return NTC_ERR_NULL_POINTER;
317 }
318
319 // Convert AdcError to ntc_err_t helper
320 auto convert_error = [](NTC::AdcError err) -> ntc_err_t {
321 switch (err) {
322 case NTC::AdcError::SUCCESS:
323 return NTC_SUCCESS;
324 case NTC::AdcError::NOT_INITIALIZED:
325 return NTC_ERR_NOT_INITIALIZED;
326 case NTC::AdcError::INVALID_CHANNEL:
327 return NTC_ERR_INVALID_PARAMETER;
328 case NTC::AdcError::READ_FAILED:
329 return NTC_ERR_ADC_READ_FAILED;
330 case NTC::AdcError::TIMEOUT:
331 return NTC_ERR_TIMEOUT;
332 case NTC::AdcError::HARDWARE_ERROR:
333 return NTC_ERR_HARDWARE_FAULT;
334 default:
335 return NTC_ERR_ADC_READ_FAILED;
336 }
337 };
338
339 // Perform sampling if configured
340 if (config_.sample_count > 1) {
341 uint64_t sum = 0;
342 uint32_t successful_samples = 0;
343
344 for (uint32_t i = 0; i < config_.sample_count; ++i) {
345 uint32_t sample_value = 0;
346 NTC::AdcError err = adc_interface_->ReadChannelCount(config_.adc_channel, &sample_value);
347 if (err == NTC::AdcError::SUCCESS) {
348 sum += sample_value;
349 successful_samples++;
350 }
351
352 // Simple delay loop if delay is configured (not accurate but hardware-agnostic)
353 if (config_.sample_delay_ms > 0 && i < config_.sample_count - 1) {
354 // Simple busy-wait delay (approximate)
355 volatile uint32_t delay_count =
356 config_.sample_delay_ms * static_cast<uint32_t>(kMillisecondsPerSecond);
357 while (delay_count-- > 0) {
358 // Busy wait
359 }
360 }
361 }
362
363 if (successful_samples == 0) {
364 return NTC_ERR_ADC_READ_FAILED;
365 }
366
367 *adc_value = static_cast<uint32_t>(sum / successful_samples);
368 return NTC_SUCCESS;
369 }
370 // Single sample
371 {
372 uint32_t sample_value = 0;
373 NTC::AdcError err = adc_interface_->ReadChannelCount(config_.adc_channel, &sample_value);
374 if (err != NTC::AdcError::SUCCESS) {
375 return convert_error(err);
376 }
377 *adc_value = sample_value;
378 return NTC_SUCCESS;
379 }
380}

718template <typename AdcType>
719ntc_err_t NtcThermistor<AdcType>::convertResistanceToTemperature(
720 float resistance_ohms, float* temperature_celsius) noexcept {
721 if (temperature_celsius == nullptr) {
722 return NTC_ERR_NULL_POINTER;
723 }
724
725 // Choose conversion method
726 switch (config_.conversion_method) {
727 case NTC_CONVERSION_LOOKUP_TABLE: {
728 const ntc_lookup_table_t* table = GetNtcLookupTable(static_cast<int>(config_.type));
729 if (table != nullptr &&
730 FindTemperatureFromLookupTable(table, resistance_ohms, temperature_celsius)) {
731 return NTC_SUCCESS;
732 }
733 // Fall back to mathematical conversion if lookup fails
734 break;
735 }
736
737 case NTC_CONVERSION_MATHEMATICAL:
738 case NTC_CONVERSION_AUTO:
739 default:
740 break;
741 }
742
743 // Use mathematical conversion (beta parameter)
744 if (!NTC::ConvertResistanceToTemperatureBeta(resistance_ohms, config_.resistance_at_25c,
745 config_.beta_value, temperature_celsius)) {
746 return NTC_ERR_CONVERSION_FAILED;
747 }
748
749 return NTC_SUCCESS;
750}

752template <typename AdcType>
753float NtcThermistor<AdcType>::applyFiltering(float new_temperature) noexcept {
754 if (!config_.enable_filtering) {
755 return new_temperature;
756 }
757
758 if (!filter_initialized_) {
759 filtered_temperature_ = new_temperature;
760 filter_initialized_ = true;
761 return new_temperature;
762 }
763
764 // Apply exponential moving average
765 filtered_temperature_ = config_.filter_alpha * new_temperature +
766 (kOneFloat - config_.filter_alpha) * filtered_temperature_;
767
768 return filtered_temperature_;
769}
