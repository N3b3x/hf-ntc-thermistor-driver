155template <typename AdcType>
156NtcError NtcThermistor<AdcType>::ReadTemperatureCelsius(
157 float *temperature_celsius) noexcept {
158 if (temperature_celsius == nullptr) {
159 return NtcError::NullPointer;
160 }
161
162 if (!initialized_) {
163 return NtcError::NotInitialized;
164 }
165
166 // Read voltage from ADC
167 float voltage_volts = 0.0F;
168 NtcError voltage_error = readAdcVoltage(&voltage_volts);
169 if (voltage_error != NtcError::Success) {
170 return voltage_error;
171 }
172
173 // Calculate resistance
174 float resistance_ohms = 0.0F;
175 NtcError resistance_error =
176 calculateResistance(voltage_volts, &resistance_ohms);
177 if (resistance_error != NtcError::Success) {
178 return resistance_error;
179 }
180
181 // Convert resistance to temperature
182 float raw_temperature = 0.0F;
183 NtcError conversion_error =
184 convertResistanceToTemperature(resistance_ohms, &raw_temperature);
185 if (conversion_error != NtcError::Success) {
186 return conversion_error;
187 }
188
189 // Apply calibration offset
190 *temperature_celsius = raw_temperature + config_.calibration_offset;
191
192 // Apply filtering if enabled
193 if (config_.enable_filtering) {
194 *temperature_celsius = applyFiltering(*temperature_celsius);
195 }
196
197 // Validate temperature range
198 if (!NTC::ValidateTemperature(*temperature_celsius, config_.min_temperature,
199 config_.max_temperature)) {
200 return NtcError::TemperatureOutOfRange;
201 }
202
203 return NtcError::Success;
204}

236template <typename AdcType>
237NtcError
238NtcThermistor<AdcType>::ReadTemperature(ntc_reading_t *reading) noexcept {
239 if (reading == nullptr) {
240 return NtcError::NullPointer;
241 }
242
243 // Initialize reading structure
244 reading->error = NtcError::Success;
245 reading->is_valid = false;
246 reading->timestamp_us = 0; // Implementer should provide timestamp
247 reading->accuracy_celsius = 0.5F; // Estimate based on typical NTC accuracy
248
249 // Read temperature
250 float temperature_celsius = 0.0F;
251 NtcError error = ReadTemperatureCelsius(&temperature_celsius);
252
253 if (error == NtcError::Success) {
254 // Get additional data
255 float resistance_ohms = 0.0F;
256 GetResistance(&resistance_ohms);
257
258 float voltage_volts = 0.0F;
259 GetVoltage(&voltage_volts);
260
261 uint32_t adc_raw = 0U;
262 GetRawAdcValue(&adc_raw);
263
264 // Fill reading structure
265 reading->temperature_celsius = temperature_celsius;
266 reading->temperature_fahrenheit = CelsiusToFahrenheit(temperature_celsius);
267 reading->temperature_kelvin = CelsiusToKelvin(temperature_celsius);
268 reading->resistance_ohms = resistance_ohms;
269 reading->voltage_volts = voltage_volts;
270 reading->adc_raw_value = adc_raw;
271 reading->is_valid = true;
272 }
273
274 reading->error = error;
275 return error;
276}

315template <typename AdcType>
316NtcError NtcThermistor<AdcType>::GetRawAdcValue(uint32_t *adc_value) noexcept {
317 if (adc_value == nullptr) {
318 return NtcError::NullPointer;
319 }
320
321 if (!initialized_) {
322 return NtcError::NotInitialized;
323 }
324
325 if (adc_interface_ == nullptr) {
326 return NtcError::NullPointer;
327 }
328
329 // Convert AdcError to NtcError helper
330 auto convert_error = [](NTC::AdcError err) -> NtcError {
331 switch (err) {
332 case NTC::AdcError::Success:
333 return NtcError::Success;
334 case NTC::AdcError::NotInitialized:
335 return NtcError::NotInitialized;
336 case NTC::AdcError::InvalidChannel:
337 return NtcError::InvalidParameter;
338 case NTC::AdcError::ReadFailed:
339 return NtcError::AdcReadFailed;
340 case NTC::AdcError::Timeout:
341 return NtcError::Timeout;
342 case NTC::AdcError::HardwareError:
343 return NtcError::HardwareFault;
344 default:
345 return NtcError::AdcReadFailed;
346 }
347 };
348
349 // Perform sampling if configured
350 if (config_.sample_count > 1) {
351 uint64_t sum = 0;
352 uint32_t successful_samples = 0;
353
354 for (uint32_t i = 0; i < config_.sample_count; ++i) {
355 uint32_t sample_value = 0;
356 NTC::AdcError err =
357 adc_interface_->ReadChannelCount(config_.adc_channel, &sample_value);
358 if (err == NTC::AdcError::Success) {
359 sum += sample_value;
360 successful_samples++;
361 }
362
363 // Simple delay loop if delay is configured (not accurate but
364 // hardware-agnostic)
365 if (config_.sample_delay_ms > 0 && i < config_.sample_count - 1) {
366 // Simple busy-wait delay (approximate)
367 volatile uint32_t delay_count =
368 config_.sample_delay_ms *
369 static_cast<uint32_t>(MILLISECONDS_PER_SECOND_);
370 while (delay_count-- > 0) {
371 // Busy wait
372 }
373 }
374 }
375
376 if (successful_samples == 0) {
377 return NtcError::AdcReadFailed;
378 }
379
380 *adc_value = static_cast<uint32_t>(sum / successful_samples);
381 return NtcError::Success;
382 }
383 // Single sample
384 {
385 uint32_t sample_value = 0;
386 NTC::AdcError err =
387 adc_interface_->ReadChannelCount(config_.adc_channel, &sample_value);
388 if (err != NTC::AdcError::Success) {
389 return convert_error(err);
390 }
391 *adc_value = sample_value;
392 return NtcError::Success;
393 }
394}

654template <typename AdcType>
655NtcError NtcThermistor<AdcType>::readAdcVoltage(float *voltage_volts) noexcept {
656 if (voltage_volts == nullptr) {
657 return NtcError::NullPointer;
658 }
659
660 if (adc_interface_ == nullptr) {
661 return NtcError::NullPointer;
662 }
663
664 // Convert AdcError to NtcError helper
665 auto convert_error = [](NTC::AdcError err) -> NtcError {
666 switch (err) {
667 case NTC::AdcError::Success:
668 return NtcError::Success;
669 case NTC::AdcError::NotInitialized:
670 return NtcError::NotInitialized;
671 case NTC::AdcError::InvalidChannel:
672 return NtcError::InvalidParameter;
673 case NTC::AdcError::ReadFailed:
674 return NtcError::AdcReadFailed;
675 case NTC::AdcError::Timeout:
676 return NtcError::Timeout;
677 case NTC::AdcError::HardwareError:
678 return NtcError::HardwareFault;
679 default:
680 return NtcError::AdcReadFailed;
681 }
682 };
683
684 // Perform sampling if configured
685 if (config_.sample_count > 1) {
686 float sum = ZERO_FLOAT_;
687 uint32_t successful_samples = 0;
688
689 for (uint32_t i = 0; i < config_.sample_count; ++i) {
690 float sample_voltage = ZERO_FLOAT_;
691 NTC::AdcError err =
692 adc_interface_->ReadChannelV(config_.adc_channel, &sample_voltage);
693 if (err == NTC::AdcError::Success) {
694 sum += sample_voltage;
695 successful_samples++;
696 }
697
698 // Simple delay loop if delay is configured (not accurate but
699 // hardware-agnostic)
700 if (config_.sample_delay_ms > 0 && i < config_.sample_count - 1) {
701 // Simple busy-wait delay (approximate)
702 volatile uint32_t delay_count =
703 config_.sample_delay_ms *
704 static_cast<uint32_t>(MILLISECONDS_PER_SECOND_);
705 while (delay_count-- > 0) {
706 // Busy wait
707 }
708 }
709 }
710
711 if (successful_samples == 0) {
712 return NtcError::AdcReadFailed;
713 }
714
715 *voltage_volts = sum / static_cast<float>(successful_samples);
716 return NtcError::Success;
717 } // Single sample
718 float sample_voltage = 0.0F;
719 NTC::AdcError err =
720 adc_interface_->ReadChannelV(config_.adc_channel, &sample_voltage);
721 if (err != NTC::AdcError::Success) {
722 return convert_error(err);
723 }
724 *voltage_volts = sample_voltage;
725 return NtcError::Success;
726}

729NtcError
730NtcThermistor<AdcType>::calculateResistance(float voltage_volts,
731 float *resistance_ohms) noexcept {

745template <typename AdcType>
746NtcError NtcThermistor<AdcType>::convertResistanceToTemperature(
747 float resistance_ohms, float *temperature_celsius) noexcept {
748 if (temperature_celsius == nullptr) {
749 return NtcError::NullPointer;
750 }
751
752 // Choose conversion method
753 switch (config_.conversion_method) {
754 case NtcConversionMethod::LookupTable: {
755 const NTC::ntc_lookup_table_t *table =
756 NTC::GetNtcLookupTable(static_cast<int>(config_.type));
757 if (table != nullptr && FindTemperatureFromLookupTable(
758 table, resistance_ohms, temperature_celsius)) {
759 return NtcError::Success;
760 }
761 // Fall back to mathematical conversion if lookup fails
762 break;
763 }
764
765 case NtcConversionMethod::Mathematical:
766 case NtcConversionMethod::Auto:
767 default:
768 break;
769 }
770
771 // Use mathematical conversion (beta parameter)
772 if (!NTC::ConvertResistanceToTemperatureBeta(
773 resistance_ohms, config_.resistance_at_25c, config_.beta_value,
774 temperature_celsius)) {
775 return NtcError::ConversionFailed;
776 }
777
778 return NtcError::Success;
779}

781template <typename AdcType>
782float NtcThermistor<AdcType>::applyFiltering(float new_temperature) noexcept {
783 if (!config_.enable_filtering) {
784 return new_temperature;
785 }
786
787 if (!filter_initialized_) {
788 filtered_temperature_ = new_temperature;
789 filter_initialized_ = true;
790 return new_temperature;
791 }
792
793 // Apply exponential moving average
794 filtered_temperature_ =
795 (config_.filter_alpha * new_temperature) +
796 ((ONE_FLOAT_ - config_.filter_alpha) * filtered_temperature_);
797
798 return filtered_temperature_;
799}

801template <typename AdcType>
802void NtcThermistor<AdcType>::initializeConfigForType(
